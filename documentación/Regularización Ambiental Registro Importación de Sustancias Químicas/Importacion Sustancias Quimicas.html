Query
 
CREATE VIEW ${nameWithSchemaName} AS SELECT DISTINCT plc.prco_id,
    plc.prco_cua,
    mcplc.chsu_id,
    mcplc.mach_id,
    dcs.dach_description,

    csd.chsd_month,
    gcc.geca_name AS "estado_declaración"

     JOIN coa_mae.project_licencing_coa plc ON ((mcplc.prco_id = plc.prco_id)))
     JOIN suia_iii.dangerous_chemistry_substances dcs ON ((mcplc.chsu_id = dcs.dach_id)))

  WHERE ((plc.prco_project_finished = true) AND (plc.prco_chemical_substances = true) AND (mcplc.mach_status = true) AND (csd.chsd_status = true))

CREATE VIEW ${nameWithSchemaName} AS SELECT DISTINCT mcplc.mach_id AS id_row,
    ( SELECT u1.user_name
           FROM users u1
          WHERE (u1.user_id = csr.user_id)) AS id_oce,
    ( SELECT p.peop_name
           FROM people p
          WHERE (p.peop_id = u.peop_id)) AS nom_oce,

    acs.achs_quantity_control AS cupo_disponible
   FROM ((((((coa_chemical_sustances.chemical_sustances_records csr
     JOIN users u ON ((u.user_id = csr.user_id)))
     JOIN coa_mae.manage_chemicals_project_licencing_coa mcplc ON (((csr.prco_id = mcplc.prco_id) AND (mcplc.mach_status IS TRUE))))
     JOIN coa_mae.project_licencing_coa plc ON (((mcplc.prco_id = plc.prco_id) AND (plc.prco_status IS TRUE))))
     JOIN suia_iii.dangerous_chemistry_substances dcs ON (((mcplc.chsu_id = dcs.dach_id) AND ((dcs.dach_type_chemistry_substance)::text = ANY (ARRAY[('SUSTANCIAS QUÍMICAS PELIGROSAS  DE TOXICIDAD AGUDA (*).'::character varying)::text, ('SUSTANCIAS QUÍMICAS PELIGROSOS DE TOXICIDAD CRÓNICA (*).'::character varying)::text])) AND ((dcs.dach_cas_code)::text = ANY (ARRAY[('7439-97-6'::character varying)::text, ('151-50-8'::character varying)::text, ('143-33-9'::character varying)::text])))))
     JOIN coa_chemical_sustances.activity_chemical_sustances acs ON (((acs.mach_id = mcplc.mach_id) AND (acs.acch_id = 1) AND (acs.achs_status IS TRUE))))
     JOIN ( SELECT t1.codigoproyecto,
            t1.processinstanceid,
            t1.nombreproceso,
            t1.estadoproceso,
            t1.fechainicioproceso,
            t1.fechafinproceso,
            t1.nombretarea,
            t1.estadotarea,
            t1.fechainiciotarea,
            t1.fechafintarea,
            t1.usuariotarea
           FROM dblink('dbname=suia_bpms_enlisy port=5632 host=172.16.0.179 user=postgres password=postgres'::text, '
								select 
								distinct on (v.value)
								v.value codigoProyecto, 
								t.processinstanceid processinstanceid, 
								p.processname as nombreProceso,
								CASE
									WHEN p.status = 1 THEN ''En trámite''
									WHEN p.status = 2 THEN ''Completado''
									WHEN p.status = 3 THEN ''Abortado''
								    WHEN p.status = 4 THEN ''Abortado''
									ELSE ''En trámite''
								END as estadoProceso, 
								p.start_date  as fechaInicioProceso ,
								p.end_date as fechaFinProceso,
								t.taskname nombreTarea,
								t.status estadoTarea ,   
								case 
									when p.status = ''1'' then coalesce(activationtime, t.createddate) 
									else t.createddate end as fechaInicioTarea, 
								case 
									when p.status = ''1'' then coalesce(logtime, t.enddate) 
									else t.enddate end as fechaFinTarea, 
								t.userid usuarioTarea  
								from variableinstancelog v
								inner join processinstancelog p on p.processinstanceid = v.processinstanceid
								inner join bamtasksummary t on t.processinstanceid = p.processinstanceid and (v.variableid=''tramite'' or v.variableid=''codigoProyecto'' or v.variableid=''numeroSolicitud'' or v.variableid=''tramite'' )
								left join task ta on p.processinstanceid =ta.processinstanceid and ta.id = t.taskid 
								left join taskevent te on te.taskid = t.taskid and te.type =''COMPLETED''
								where 
								user_identity not like ''%msit%''
								and user_identity not like ''%admin%''
							        and p.processname in (''Registro Sustancias Quimicas'', ''Registro Sustancias Quimicas (Importacion Exportacion)'')
							        and p.status in (2)  
 								and t.status not in (''Exited'')
								order by 1, 9 desc'::text) t1(codigoproyecto text, processinstanceid text, nombreproceso text, estadoproceso text, fechainicioproceso timestamp without time zone, fechafinproceso timestamp without time zone, nombretarea text, estadotarea text, fechainiciotarea timestamp without time zone, fechafintarea timestamp without time zone, usuariotarea text)
          GROUP BY t1.codigoproyecto, t1.processinstanceid, t1.nombreproceso, t1.estadoproceso, t1.fechainicioproceso, t1.fechafinproceso, t1.nombretarea, t1.estadotarea, t1.fechainiciotarea, t1.fechafintarea, t1.usuariotarea
          ORDER BY t1.codigoproyecto) y ON ((y.codigoproyecto = (plc.prco_cua)::text)))
  WHERE (date_part('year'::text, csr.chsr_valid_until) = ( SELECT date_part('year'::text, now()) AS date_part))
