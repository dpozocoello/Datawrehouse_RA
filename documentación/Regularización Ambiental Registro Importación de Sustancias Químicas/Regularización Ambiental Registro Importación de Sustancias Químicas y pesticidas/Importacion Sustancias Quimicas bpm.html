CREATE VIEW ${nameWithSchemaName} AS SELECT c.table_schema,
    c.table_name,
    c.column_name,
    pgd.description
   FROM ((pg_statio_all_tables st
     JOIN pg_description pgd ON ((pgd.objoid = st.relid)))
     RIGHT JOIN information_schema.columns c ON (((pgd.objsubid = (c.ordinal_position)::integer) AND ((c.table_schema)::name = st.schemaname) AND ((c.table_name)::name = st.relname))))

CREATE OR REPLACE FUNCTION public.corregir_fecha_fin_tareas_completas()
 RETURNS void
 LANGUAGE plpgsql
AS $function$
DECLARE

var_fecha_fin_anterior timestamp;
var_processinstanceid int := 0;
var_userid_anterior varchar(255) := '';
reg record;
BEGIN
         
        FOR REG IN (select id, processinstanceid 
			from task where status = 'Completed' and id not in (select taskid from taskevent where type = 'COMPLETED')
			and processinstanceid in (select processinstanceid from variableinstancelog where value like 'MAE-RA%' and variableid in ('tramite','codigoProyecto'))
			and processinstanceid not in (select processinstanceid from processinstancelog where status in (2, 3, 4))
			--and processinstanceid = 8549
			order by processinstanceid, createdon
		   ) LOOP
		RAISE INFO 'Tarea: %',reg.id;
		--selecciona siguiente tarea
		select id into var_task_id_anterior from task where processinstanceid = reg.processinstanceid and id > reg.id
		order by 1 limit 1;
		--selecciona usuario del evento added
		select userid into var_userid_anterior from taskevent where taskid = reg.id and type='ADDED' limit 1;
		RAISE INFO 'Tarea siguiente: %',var_task_id_anterior;
		if var_task_id_anterior is not null then
			--selecciona la fecha added de la tarea siguiente
			select logtime into var_fecha_fin_anterior
				from taskevent
				where taskid = var_task_id_anterior
				and type='ADDED';
			RAISE INFO 'fecha siguiente: %',var_fecha_fin_anterior;
			--inserta si la fecha no es nula.
			If var_fecha_fin_anterior is not null then
				--inserta
				INSERT INTO taskevent(id, logtime, taskid, type, userid, optlock)
				VALUES (nextval('task_event_id_seq'),var_fecha_fin_anterior, reg.id, 'COMPLETED', var_userid_anterior, 0); 
			else
				RAISE INFO 'No Cambia, fecha nula';
			end if;
		else
			RAISE INFO 'No tiene tarea siguiente';
		end if;
	END LOOP;
END;

 public.datediff   
 
CREATE OR REPLACE FUNCTION public.datediff(difftype character varying, date1 date, date2 date)
 RETURNS integer
 LANGUAGE plpgsql
AS $function$
DECLARE
	YEAR_CONST Character Varying(15) := 'year';
	MONTH_CONST Character Varying(15) := 'month';
	DAY_CONST Character Varying(15) := 'day';
 
	diffInInterval Interval;
	diffInDoublePrecision Double Precision := 0;
	diffInInteger Integer := 0;
	dateTemp Date;
BEGIN
 
	diffInInterval := age(date2, date1);
 
	IF lower($1) = lower(YEAR_CONST) THEN
		diffInDoublePrecision := date_part('Year', diffInInterval);
	ELSEIF lower($1) = lower(MONTH_CONST) THEN
		diffInDoublePrecision := (date_part('Year', diffInInterval) * 12) + date_part('Month', diffInInterval);
	ELSEIF lower($1) = lower(DAY_CONST) THEN
		diffInDoublePrecision := date2 - date1;
	END IF;
 
	diffInInteger := CAST(diffInDoublePrecision AS Integer);
	RETURN diffInInteger;
END;
$function$
;

CREATE OR REPLACE FUNCTION public.eliminar_procesos_duplicados()
 RETURNS void
 LANGUAGE plpgsql
AS $function$

tmp_count int = 0;

					(select id from tmp_tareas)

				where deadlines_startdeadline_id in
					(select id from tmp_tareas)

					(select id from tmp_tareas)

					(select id from tmp_tareas)

					(select id from tmp_tareas)

			RAISE INFO 'notification_bas: %', tmp_count;

				select id from i18ntext 
				where task_descriptions_id in (select id from tmp_tareas)
				or task_subjects_id in (select id from tmp_tareas)

				select id from i18ntext

				or task_subjects_id in (select id from tmp_tareas)

				select id from i18ntext where 
				notification_names_id in (
					select id from notification where escalation_notifications_id in 
					(select id from deadline 
						where deadlines_startdeadline_id in
							(select id from tmp_tareas)
						or deadlines_enddeadline_id in
							(select id from tmp_tareas)
					)
				)
				or notification_subjects_id in (
					select id from notification where escalation_notifications_id in 
					(select id from deadline 
						where deadlines_startdeadline_id in
							(select id from tmp_tareas)
						or deadlines_enddeadline_id in
							(select id from tmp_tareas)
					)
				)
			);
			RAISE INFO 'i18ntext2: %', tmp_count;
			
			--eliminamos de notification
			select count(*) into tmp_count from notification where escalation_notifications_id in 
			(select id from deadline 
				where deadlines_startdeadline_id in
					(select id from tmp_tareas)
				or deadlines_enddeadline_id in
					(select id from tmp_tareas)
			);

			(select id from deadline 
				where deadlines_startdeadline_id in
					(select id from tmp_tareas)
				or deadlines_enddeadline_id in
					(select id from tmp_tareas)
			);

			select count(*) into tmp_count from escalation where deadline_escalation_id in
			(select id from deadline 
				where deadlines_startdeadline_id in
					(select id from tmp_tareas)
				or deadlines_enddeadline_id in
					(select id from tmp_tareas)
			);

			(select id from deadline 
				where deadlines_startdeadline_id in
					(select id from tmp_tareas)
				or deadlines_enddeadline_id in
					(select id from tmp_tareas)
			);

			or deadlines_enddeadline_id in

				update task set createdon = var_start_date, activationtime = var_start_date where id = var_task_id;
				update bamtasksummary set createddate = var_start_date where taskid = var_task_id;
				update taskevent set logtime = var_start_date where taskid = var_task_id and type = 'ADDED';
			else
				RAISE INFO 'No hay codigo de tarea';
			end if ; 
			
			
		else
			RAISE INFO 'ERROR FECHA INICIO NULL!!!';
		end if;
		-- elimino la tabla temporal
		drop table tmp_procesos;
		drop table tmp_tareas;
	END LOOP;
END;

 public.func_abort_update   
 
CREATE OR REPLACE FUNCTION public.func_abort_update(integer, integer)
 RETURNS integer
 LANGUAGE plpgsql
AS $function$
BEGIN
-- $1 id proceso antiguo 
-- $2 id proceso nuevo
--ACTUALIZAR PROCESSINSTALOG
update processinstancelog set start_date=(select start_date from processinstancelog where processinstanceid=$1) where id=$2;
update processinstancelog set status=4 where id=$1;
--ACTUALIZAR TASK
update task set activationtime=a.createddate,createdon=a.createddate from
(
select * from (select ROW_NUMBER() OVER (PARTITION BY processinstanceid) AS item,* 
from (select pk,createddate,enddate,startdate,processinstanceid from bamtasksummary  
where processinstanceid in ( $1) order by pk asc)z)z
inner join 
(select ROW_NUMBER() OVER (PARTITION BY processinstanceid) AS item,* 
from (select id,processinstanceid from task  
where processinstanceid in ( $2) order by id asc)z)y
on z.item=y.item
)a where task.id=a.id;
--ACTUALIZAR bamtasksummary
update bamtasksummary set createddate=a.createddate,enddate=a.enddate,startdate=a.startdate from
(
select * from (select ROW_NUMBER() OVER (PARTITION BY processinstanceid) AS item,* 
from (select createddate,enddate,startdate,processinstanceid from bamtasksummary  
where processinstanceid in ( $1) order by pk asc)z)z
inner join 
(select ROW_NUMBER() OVER (PARTITION BY processinstanceid) AS item,PK 
from (select pk,createddate,enddate,startdate,processinstanceid from bamtasksummary  
where processinstanceid in ( $2) order by pk asc)z)y
on z.item=y.item
)a where bamtasksummary.pk=a.pk;
--ACTUALIZAR event
update taskevent set logtime=x.enddate from (select (case when enddate is null  then now() else enddate end) as enddate, taskid from bamtasksummary where processinstanceid=$1)x
where taskevent.taskid=x.taskid and type='COMPLETED';

RETURN 1;
END;
$function$

 public.liberamemoria   
 
CREATE OR REPLACE FUNCTION public.liberamemoria()
 RETURNS text
 LANGUAGE plpgsql
AS $function$ 
DECLARE
 resultado_txt TEXT;
BEGIN
   WITH inactive_connections AS (
    SELECT
        pid,
        rank() over (partition by client_addr order by backend_start ASC) as rank
    FROM 
        pg_stat_activity
    WHERE
        -- Exclude the thread owned connection (ie no auto-kill)
        pid <> pg_backend_pid( )
    AND
        -- Exclude known applications connections
        application_name !~ '(?:psql)|(?:pgAdmin.+)'
    AND
        -- Include connections to the same database the thread is connected to
        datname = current_database() 
    AND
        -- Include connections using the same thread username connection
        usename = current_user 
    AND
        -- Include inactive connections only
        state in ('idle', 'idle in transaction', 'idle in transaction (aborted)', 'disabled') 
    AND
        -- Include old connections (found with the state_change field)
        current_timestamp - state_change > interval '5 minutes' 
)
SELECT
    pg_terminate_backend(pid)
FROM
    inactive_connections 
WHERE
    rank > 1 ;
   resultado_txt :=  'Se relaizó la desactivación del proceso de registro ambiental';
 
 RETURN resultado_txt;
END;
$function$

CREATE OR REPLACE FUNCTION public.obtener_reporte_proyecto(p_codigo text)
 RETURNS TABLE(codigoproyecto text, processinstanceid text, nombreproceso text, estadoproceso text, fechainicioproceso timestamp without time zone, fechafinproceso timestamp without time zone, nombretarea text, estadotarea text, fechainiciotarea timestamp without time zone, fechafintarea timestamp without time zone, usuariotarea text, "NOMBRE USUARIO TAREA" text, motivo text)
 LANGUAGE plpgsql
AS $function$
BEGIN
  RETURN QUERY
  WITH proyecto_base AS (
    SELECT v.value::text AS cod_proy_base
    FROM variableinstancelog v
    INNER JOIN processinstancelog p ON p.processinstanceid = v.processinstanceid
    WHERE v.value IN (
        SELECT pren_id::text

            'SELECT pren_id FROM suia_iii.projects_environmental_licensing WHERE pren_code = ''' || p_codigo || ''' OR pren_id::text = ''' || p_codigo || ''''
        ) AS t(pren_id int)
        UNION

            'SELECT prco_id FROM coa_mae.project_licencing_coa WHERE prco_cua = ''' || p_codigo || ''' OR prco_id::text = ''' || p_codigo || ''''
        ) AS t(prco_id int)
    )
    LIMIT 1
  ),
  valor_busqueda AS (
    SELECT cod_proy_base FROM proyecto_base
  ),
  project_licencing_coa_ext AS (
    SELECT *
    FROM dblink(
        'dbname=suia_enlisy port=5632 host=172.16.0.179 user=postgres password=postgres',

         FROM coa_mae.project_licencing_coa 
         WHERE prco_cua = ''' || p_codigo || ''' OR prco_id::text = ''' || p_codigo || ''''
    ) AS t(prco_id int, prco_cua text, prco_delete_reason text, prco_deactivation_date timestamp, prco_creator_user text)
  ),
  projects_environmental_ext AS (
    SELECT *
    FROM dblink(
        'dbname=suia_enlisy port=5632 host=172.16.0.179 user=postgres password=postgres',
        'SELECT pren_id, pren_code, pren_delete_reason, pren_deactivation_date, pren_creator_user

            WHEN p.status = 1 THEN 'En trámite'
            WHEN p.status = 2 THEN 'Completado'
            WHEN p.status = 3 THEN 'Abortado'
            WHEN p.status = 4 THEN 'Abortado'
            WHEN p.status = 5 THEN 'Abortado 5'
            ELSE 'Abortado'

        p.start_date AS fecha_ini_proc,
        p.end_date AS fecha_fin_proc,
        t.taskname AS nom_tarea,

        ''::text AS nombre_usuario_tarea,
        NULL::text AS motivo
    FROM variableinstancelog v
    LEFT JOIN processinstancelog p ON p.processinstanceid = v.processinstanceid
    LEFT JOIN bamtasksummary t ON t.processinstanceid = p.processinstanceid
    LEFT JOIN task ta ON p.processinstanceid = ta.processinstanceid AND ta.id = t.taskid

        NULL::text AS estado_tarea,
        NULL::timestamp AS fecha_ini_tarea,
        NULL::timestamp AS fecha_fin_tarea,
        b.pren_creator_user AS usuario_tarea,
        ''::text AS nombre_usuario_tarea,
        b.pren_delete_reason AS motivo
    FROM projects_environmental_ext b
    WHERE NOT EXISTS (
      SELECT 1 FROM reporte_union q WHERE q.cod_proy_union = b.pren_id::text)
    UNION ALL
    SELECT 
        a.prco_id::text AS cod_proy_union,
        NULL::text AS proc_inst_id,
        NULL::text AS nombre_proceso,
        'Sin proceso'::text AS estado_proceso,
        NULL::timestamp AS fecha_ini_proc,
        a.prco_deactivation_date AS fecha_fin_proc,
        NULL::text AS nom_tarea,
        NULL::text AS estado_tarea,
        NULL::timestamp AS fecha_ini_tarea,
        NULL::timestamp AS fecha_fin_tarea,
        a.prco_creator_user AS usuario_tarea,
        ''::text AS nombre_usuario_tarea,
        a.prco_delete_reason AS motivo
    FROM project_licencing_coa_ext a
    WHERE NOT EXISTS (
      SELECT 1 FROM reporte_union q WHERE q.cod_proy_union = a.prco_id::text)
  )
SELECT 
    ru.cod_proy_union::text AS codigoProyecto,
    ru.proc_inst_id::text AS processinstanceid,
    ru.nombre_proceso::text AS nombreProceso,
    ru.estado_proceso::text AS estadoProceso,
    ru.fecha_ini_proc AS fechaInicioProceso,
    ru.fecha_fin_proc AS fechaFinProceso,
    ru.nom_tarea::text AS nombreTarea,
    ru.estado_tarea::text AS estadoTarea,
    ru.fecha_ini_tarea AS fechaInicioTarea,
    ru.fecha_fin_tarea AS fechaFinTarea,
    ru.usuario_tarea::text AS usuarioTarea,
    ru.nombre_usuario_tarea::text AS "NOMBRE USUARIO TAREA",
    ru.motivo::text AS motivo
FROM reporte_union ru
UNION ALL
SELECT 
    sp.cod_proy_union::text AS codigoProyecto,
    sp.proc_inst_id::text AS processinstanceid,
    sp.nombre_proceso::text AS nombreProceso,
    sp.estado_proceso::text AS estadoProceso,

END;

 public.search_columns   
 
CREATE OR REPLACE FUNCTION public.search_columns(needle text, haystack_tables name[] DEFAULT '{}'::name[], haystack_schema name[] DEFAULT '{public}'::name[])
 RETURNS TABLE(schemaname text, tablename text, columnname text, rowctid text)
 LANGUAGE plpgsql
AS $function$
begin
  FOR schemaname,tablename,columnname IN
      SELECT c.table_schema,c.table_name,c.column_name
      FROM information_schema.columns c
      JOIN information_schema.tables t ON
        (t.table_name=c.table_name AND t.table_schema=c.table_schema)
      WHERE (c.table_name=ANY(haystack_tables) OR haystack_tables='{}')
        AND c.table_schema=ANY(haystack_schema)
        AND t.table_type='BASE TABLE'
  LOOP
    EXECUTE format('SELECT ctid FROM %I.%I WHERE cast(%I as text)=%L',
       schemaname,
       tablename,
       columnname,
       needle
    ) INTO rowctid;
    IF rowctid is not null THEN
      RETURN NEXT;
    END IF;
 END LOOP;
END;
$function$
;

 public.workload2   
 
CREATE OR REPLACE FUNCTION public.workload2(usuarios text)
 RETURNS SETOF record
 LANGUAGE plpgsql
AS $function$
DECLARE
	r record;
BEGIN    
	CREATE TEMPORARY TABLE tempUsers AS SELECT unnest(string_to_array(usuarios, ',')) AS usuarios;
	FOR r IN SELECT CASE WHEN A.numero_tareas IS NULL THEN 0
					ELSE A.numero_tareas
				       END
				      ,B.usuarios 
				FROM
				(
					SELECT  count(*) AS numero_tareas, 
						actualowner_id AS usuario
					FROM task WHERE status NOT IN ('Completed', 'Failed')
					AND actualowner_id IN (SELECT unnest(string_to_array(usuarios, ',')))  
					GROUP BY actualowner_id 
					ORDER BY 1,2
				) AS A 
				RIGHT JOIN tempUsers B
				ON (a.usuario=b.usuarios) 
	LOOP
		RETURN NEXT r ;
	END LOOP;
	DROP TABLE tempUsers;
	RETURN;
	
END;
$function$
;
